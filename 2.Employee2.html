1. Increase the salary of all employees in the 'Sales' department by 5%. 
db.employee.updateMany(
    { department: 'Sales' },
    { $mul: { salary: 1.05 } }
);
2. Update the salary of the employee with id: 10 to increase it by 10%. 
db.employee.updateOne(
    { id: 10 },
    { $mul: { salary: 1.10 } }
);
3. Set the minimum salary to 35000 for all employees in 'Engineering' with salary below this value. 
db.employee.updateMany(
    { department: 'Engineering', salary: { $lt: 35000 } },
    { $set: { salary: 35000 } }
);
4. Ensure salary of employee with id: 5 is not more than 32000. 
db.employee.updateOne(
    { id: 5, salary: { $gt: 32000 } },
    { $set: { salary: 32000 } }
);
5. Remove the email field from all employees in the 'Marketing' department. 
db.employee.updateMany(
    { department: 'Marketing' },
    { $unset: { email: "" } }
);
6. Double the salary of the employee with id: 3. 
db.employee.updateOne(
    { id: 3 },
    { $mul: { salary: 2 } }
);
7. Rename field 'department' to 'division' for employee with id: 2. 
db.employee.updateOne(
    { id: 2 },
    { $rename: { 'department': 'division' } }
);
8. Insert a new employee with id: 21 (Alexa) if not exists. 
db.employee.updateOne(
    { id: 21 },
    { $setOnInsert: { id: 21, name: 'Alexa', gender: 'Female', salary: 55000, department: 'IT' } },
    { upsert: true }
);
9. Add 'projectAlpha' to a new array field 'projects' for employee with id: 4. 
db.employee.updateOne(
    { id: 4 },
    { $push: { projects: 'projectAlpha' } }
);
10. Add multiple projects ('projectBeta', 'projectGamma') to 'projects' array for employee with id: 7. 
db.employee.updateOne(
    { id: 7 },
    { $push: { projects: { $each: ['projectBeta', 'projectGamma'] } } }
);
11. Remove the last project from 'projects' array for employee with id: 4. 
db.employee.updateOne(
    { id: 4 },
    { $pop: { projects: 1 } }
);
12. Remove the first project from 'projects' array for employee with id: 7. 
db.employee.updateOne(
    { id: 7 },
    { $pop: { projects: -1 } }
);
13. Remove 'projectBeta' from 'projects' array wherever it exists. 
db.employee.updateMany(
    { projects: 'projectBeta' },
    { $pull: { projects: 'projectBeta' } }
);
14. Find the employee with id = 5. 
db.employee.findOne({ id: 5 });
15. Find the employee whose department is exactly 'Business Development'. 
db.employee.findOne({ department: 'Business Development' });
16. Find all employees whose gender is not 'Male'. 
db.employee.find({ gender: { $ne: 'Male' } });
17. Find all documents where department is not 'Sales'. 
db.employee.find({ department: { $ne: 'Sales' } });
18. Find all employees whose salary is greater than 50000. 
db.employee.find({ salary: { $gt: 50000 } });
19. Find all employees with id greater than 15. 
db.employee.find({ id: { $gt: 15 } });
20. Find all employees whose salary is greater than or equal to 80000. 
db.employee.find({ salary: { $gte: 80000 } });
